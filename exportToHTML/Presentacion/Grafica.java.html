415      public void getDec(double[] x1) { 
416          x = new double[x1.length]; 
417          y = new double[x1.length]; 
418          z = new double[x1.length][x1.length]; 
419           
420          for(int i = 0; i < x.length; i++){ 
421              x[i]=i+1; 
422          } 
423          for (int i = 0; i < y.length; i++) { 
424              y[i] = x1[i]; 
425               
426          } 
427           
428          for (int i = 0; i < z.length; i++) { 
429              for(int j=0;j<z.length;j++){ 
430                  z[j][i]=x1[i]; 
431              } 
432               
433          } 
434      }
